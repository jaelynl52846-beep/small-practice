    <script>
        // 显示/隐藏预处理选项
        const needPreprocess = document.getElementById('need_preprocess');
        const ratioOptions = document.getElementById('ratio_options');
        const customSize = document.getElementById('custom_size');
        const ratioChoice = document.getElementById('ratio_choice');

        needPreprocess.addEventListener('change', function() {
            ratioOptions.style.display = this.checked ? 'block' : 'none';
        });

        // 显示/隐藏自定义尺寸
        ratioChoice.addEventListener('change', function() {
            customSize.style.display = this.value === '7' ? 'block' : 'none';
        });

        // 初始化检查
        if (ratioChoice.value === '7') {
            customSize.style.display = 'block';
        }

        // 新增：退出确认逻辑
        function confirmExit() {
            if (confirm('确定要退出程序吗？退出后将无法继续处理图片！')) {
                // 发送退出请求
                fetch('/exit')
                    .then(response => response.json())
                    .then(data => {
                        // 关闭当前网页
                        window.close();
                    })
                    .catch(error => {
                        // 即使请求失败，也关闭网页
                        window.close();
                    });
            }
        }
    </script>
